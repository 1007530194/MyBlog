def get_emb(embedding,string_tensor, delimiter='_'):
    
    batch_split = tf.string_split(string_tensor, delimiter='_')
    batch_tonum = tf.string_to_number(batch_split.values, out_type=tf.int32)
    
    batch_mats = tf.sparse_to_dense(batch_split.indices,batch_split.dense_shape, batch_tonum,0)
    
    # 标记，大于等于1的
    batch_mask = tf.to_float(batch_mats>0)
    batch_nums = tf.reduce_sum(batch_mask,axis=1)
    
    batch_mask = tf.reshape(batch_mask, tf.concat([tf.shape(batch_mats), [1]],0))
    batch_nums = tf.reshape(batch_nums, tf.concat([tf.shape(batch_nums), [1]],0))
    
    batch_embedding = tf.nn.embedding_lookup(embedding, batch_mats)
    input_itembuy = tf.reduce_sum(tf.multiply(batch_embedding, batch_mask),1)
    input_itembuy = tf.div(input_itembuy, batch_nums)

    return input_itembuy

def train_fun():
    print_debug("begin Employee")

    batch_labels = tf.placeholder(tf.int64, [None, 1])

    # -------------------------------------------------------------------------------------------
    # 商品特征, emb_mask标记每条记录的商品, word_num记录商品个数, batch_itembuy记录商品
    emb_mask = tf.placeholder(tf.float32, shape=[None, max_window_size, 1])
    batch_item_num = tf.placeholder(tf.float32, shape=[None, 1])

    batch_item_buy = tf.placeholder(tf.int32, shape=[None, max_window_size])
    input_emb_itembuy = tf.nn.embedding_lookup(embedding['item_embedding'], batch_item_buy)
    input_itembuy = tf.div(tf.reduce_sum(tf.multiply(input_emb_itembuy, emb_mask), 1), batch_item_num)

    nonzero = tf.count_nonzero(batch_item_buy + 1)
    tf.greater(batch_item_buy, 1)

    # 用户特征
    batch_user_feature = tf.placeholder(tf.float32, shape=[None, user_feature_size])

    # 输入网络的特征,有商品特征和用户特征构成
    input_net = tf.concat([input_itembuy, batch_user_feature], 1)
    # -------------------------------------------------------------------------------------------
    # 进入DNN三层网络，得到最后一层隐层,也是softmax层的输入层，可作为用户向量
    softmax_input = multilayer_perceptron(input_net, weights, biases)

    # softmax层 权重和阈值
    nce_weights = tf.Variable(tf.truncated_normal([item_size, n_hidden_1], stddev=1.0 / math.sqrt(n_hidden_1)))
    nce_biases = tf.Variable(tf.zeros([item_size]))

    # 整体的训练进度
    global_step = tf.Variable(0, name='global_step', trainable=False)

    # 计算损失函数
    loss = tf.reduce_mean(tf.nn.nce_loss(weights=nce_weights,
                                         biases=nce_biases,
                                         labels=batch_labels,
                                         inputs=softmax_input,
                                         num_sampled=10,
                                         num_classes=item_size))

    cost = tf.reduce_sum(loss) / batch_size
    # 优化器
    optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate).minimize(cost, global_step=global_step)

    # -----------------------------------------------------------------------------------------
    # 计算输出层
    out_layer = tf.matmul(softmax_input, tf.transpose(nce_weights)) + nce_biases
    similarity, sim_indices = tf.nn.top_k(tf.nn.softmax(out_layer), 100)

    # 向量输出
    item_weight = tf.transpose(weights['out'])
    # -----------------------------------------------------------------------------------------

    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())

        step = 0
        while True:
            # 获取单批次数据
            batch_list = next_batch(batch_size)
            user_id, buy_item, user_feature, label, batch_mask, word_number = next_batch(batch_list)

            # 最大迭代次数
            if step > max_step:
                break

            # 传入网络的参数数据
            feed_dict = {batch_item_buy: buy_item,
                         emb_mask: batch_mask,
                         batch_item_num: word_number,
                         batch_labels: label,
                         batch_user_feature: user_feature}

            _, cost_val, user_vec = sess.run([optimizer, cost, softmax_input], feed_dict=feed_dict)

            print_debug("cost: {0}\t step={1}".format(cost_val, step))
            step += 1

        # 店铺向量
        item_vec = sess.run(item_weight)
        print_debug("item_vec  {0}".format(item_vec.shape))
        for key in item_dict:
            item_feature[key] = item_vec[item_dict[key]]

import tensorflow as tf
import numpy as np

# Create 100 phony x, y data points in NumPy, y = x * 0.1 + 0.3
x_data = np.random.rand(100).astype(np.float32)
y_data = x_data * 0.1 + 0.3

# Try to find values for W and b that compute y_data = W * x_data + b
# (We know that W should be 0.1 and b 0.3, but TensorFlow will
# figure that out for us.)
W = tf.Variable(tf.random_uniform([1], -1.0, 1.0))
b = tf.Variable(tf.zeros([1]))
y = W * x_data + b

# Minimize the mean squared errors.
loss = tf.reduce_mean(tf.square(y - y_data))
optimizer = tf.train.GradientDescentOptimizer(0.5)
train = optimizer.minimize(loss)

# Before starting, initialize the variables.  We will 'run' this first.
init = tf.global_variables_initializer()

# Launch the graph.
sess = tf.Session()
sess.run(init)

# Fit the line.
for step in range(201):
    sess.run(train)
    if step % 20 == 0:
        print(step, sess.run(W), sess.run(b))

# Learns best fit is W: [0.1], b: [0.3]
